-- STAT_0002_0004_compare_date
-- <#[ФК_МесОтч_Доходы]#>
with
    member [Measures].[ДанныеНа]
    as '[Период].[Период].CurrentMember.UniqueName'
    set [NonEmptySet]
    as 'Tail
        (
            Filter
            (
                [Период].[Период].Members,
                not IsEmpty([Measures].[Исполнено])
            ),
            1
        )'

    { [Measures].[ДанныеНа] } on columns,
    { [NonEmptySet] } on rows
from  [ФК_МесОтч_Доходы]
  ]]>
  </query>

-- STAT_0002_0004_compare_Grid  

SELECT
    NON EMPTY [Период].[Период].[День].Members DIMENSION PROPERTIES MEMBER_UNIQUE_NAME ON ROWS ,
NON EMPTY    {
        Descendants 
        (
            [Показатели].[Мониторинг СЭР].[Все показатели].[Отраслевая структура безработных граждан] ,

            {
                [Показатели].[Мониторинг СЭР].[Все показатели].[Отраслевая структура безработных граждан]
            } 
        ,
        Descendants   
        (
            [Показатели].[Мониторинг СЭР].[Все показатели].[Отраслевая структура безработных граждан].DataMember ,
            [Показатели].[Мониторинг СЭР].[Все показатели].[Отраслевая структура безработных граждан].DataMember.Level ,
            SELF_AND_AFTER

                [Показатели].[Мониторинг СЭР].[Все показатели].[Отраслевая структура безработных граждан]
            } 
        
    }      ON COLUMNS

    (
        [Источники данных].[Все источники данных].[СТАТ Отчетность - Субъект РФ] ,
        [Территории].[РФ].[Все территории].[Российская Федерация].<%region%>

          {
            Descendants

             ,[ОК].[ОКВЭД].[Все виды экономической деятельности].Level

          [ОК].[ОКВЭД].[Раздел].ALLMEMBERS

        }
      )} ON COLUMNS

  ([Measures].[За период],
    [Источники данных].[Все источники данных].[СТАТ Отчетность - ФСГС]
   ,[Группировки].[Труд_Трудовые ресурсы].[Все группировки].[По видам экономической деятельности]
   ,[Территории].[РФ].[Все территории].[Российская Федерация].<%region%>
   ,[Труд].[Трудовые ресурсы].[Все показатели].[Среднегодовая численность занятого населения]
  )

   -- STAT_0002_0004_Dates

    Set [Непустые периоды]
    as 'Filter
        (
            Descendants
            (
                [Период].[Период].[Данные всех периодов],
                [Период].[Период].[День],
                SELF
            ),
            [Measures].[Значение] <> 0
        )'

    {[Measures].[Значение]} ON COLUMNS ,
    NON EMPTY {[Непустые периоды] - {[Период].[Период].[Данные всех периодов].[2010].[Полугодие 1].[Квартал 2].[Апрель].[8]}} DIMENSION PROPERTIES MEMBER_UNIQUE_NAME ON ROWS

    (
        [Источники данных].[Все источники данных].[СТАТ Отчетность - Субъект РФ] ,
        [Территории].[РФ].[Все территории].[Российская Федерация].<%region%>,
        [Показатели].[Мониторинг СЭР].[Все показатели].[Отраслевая структура безработных граждан]
