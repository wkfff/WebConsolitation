  <query id="last_year">
    <![CDATA[ -- name: last_year   
SELECT
    { [Measures].[Значение]  } on columns,
    {
        Tail  
        (
            Filter  
            (
                [Период].[Год].Members,
                [Measures].[Значение] > 0  
            )  
        ).Item(0)  
    } DIMENSION PROPERTIES MEMBER_UNIQUE_NAME on rows  
FROM [МО_Оценка ОМСУ_Оценка эффективности]  
where  
    (
        <%baseRegion%>,
        [Источники данных].[Все источники данных].[МО Оценка ОМСУ],
        [ОМСУ].[Оценка эффективности].[Все показатели]
    )      
	  ]]>
  </query>
