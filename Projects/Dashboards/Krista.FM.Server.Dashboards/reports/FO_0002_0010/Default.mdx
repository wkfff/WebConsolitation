with
    member Measures.[ДанныеНа]
    as '<%period_dimension%>.CurrentMember.UniqueName'

    {Measures.[ДанныеНа]} on columns,

        Filter
        (
            OpeningPeriod

   [Уровни бюджета].[СКИФ].[Все]
