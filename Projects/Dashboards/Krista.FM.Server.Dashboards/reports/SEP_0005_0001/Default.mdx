  <query id="SEP_0005_0001_date">
    <![CDATA[ -- SEP_0005_0001_date
with
    set [Годы]
    as 'filter
        (
            [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].[Год].Members,
            not IsEmpty([Measures].[Значение])
        )'
    member [Measures].[Период]
    as '[Период__Год Квартал Месяц].[Период__Год Квартал Месяц].CurrentMember.UniqueName'
select
    [Measures].[Период] on columns,
    [Годы] on rows
from [ЭО_Мониторинг СЭП_МесМониторинг]
where
    (
        [СЭП__МесМониторинг].[СЭП__МесМониторинг].[Все показатели],
        [Территории__Сопоставимый].[Территории__Сопоставимый].[Все территории].[Российская Федерация].[Уральский федеральный округ].[Ханты-Мансийский автономный округ]
    )
    ]]>
  </query>

  <query id="SEP_0005_0001_grid">
    <![CDATA[ -- SEP_0005_0001_grid
with
    set [Месяцы]
    as 'Filter
        (
            Descendants
            (
                [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].[Данные всех периодов].[<%selected_year%>],
                [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].[Месяц]
            ),
            not [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].CurrentMember is [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].CurrentMember.Parent.DataMember
        )'
    set [Параметры]
    as 'Filter
        (
            Descendants
            (
                [СЭП__МесМониторинг].[СЭП__МесМониторинг].[Все показатели],
                1,
                SELF_AND_AFTER
            ),
            not [СЭП__МесМониторинг].[СЭП__МесМониторинг].CurrentMember is [СЭП__МесМониторинг].[СЭП__МесМониторинг].CurrentMember.Parent.DataMember
        ) - [СЭП__МесМониторинг].[СЭП__МесМониторинг].[Все показатели].[Индекс промышленного производства]'
    member [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].[Данные всех периодов].[Показатели]
    as 'null'
    member [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].[Данные всех периодов].[Единица измерения]
    as 'null'
    member [Measures].[Показатели]
    as '[СЭП__МесМониторинг].[СЭП__МесМониторинг].CurrentMember.Properties("Краткое наименование")'
    member [Measures].[Единица измерения]
    as 'iif([СЭП__МесМониторинг].[СЭП__МесМониторинг].CurrentMember.Level.Name = "Уровень 1",
    "Уровень 1",
    [СЭП__МесМониторинг].[СЭП__МесМониторинг].CurrentMember.Properties("Примечание"))'
    member [Measures].[Данные]
    as 'iif
        (
            [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].CurrentMember is [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].[Данные всех периодов].[Единица измерения],
            [Measures].[Единица измерения],
            iif
            (
                [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].CurrentMember is [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].[Данные всех периодов].[Показатели],
                [Measures].[Показатели],
                [Measures].[Значение]
            )
        )'
select
    {
        {
            [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].[Данные всех периодов].[Показатели],
            [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].[Данные всех периодов].[Единица измерения]
        } + [Месяцы]
    } on columns,
    [Параметры] on rows
from [ЭО_Мониторинг СЭП_МесМониторинг]
where
    (
        [Территории__Сопоставимый].[Территории__Сопоставимый].[Все территории].[Российская Федерация].[Уральский федеральный округ].[Ханты-Мансийский автономный округ],
        [Measures].[Данные]
    )
    ]]>
  </query>
