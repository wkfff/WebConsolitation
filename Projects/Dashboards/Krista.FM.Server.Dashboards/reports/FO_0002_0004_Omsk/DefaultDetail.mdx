                        [КД].[Сопоставимый].[НДФЛ ],
                        [Measures].[<%fact_measure%>]
                    ) +
                    (
                        [КД].[Сопоставимый].[НДФЛ ],
                        [Measures].[Факт_за период ]
                    ),
                    IIF
                    (
                        [КД].[Сопоставимый].CurrentMember is [КД].[Сопоставимый].[Налоговые доходы ],
                        (
                            [КД].[Сопоставимый].[Налоговые доходы ],
                            [Measures].[<%fact_measure%>]
                        ) -
                        (
                            [КД].[Сопоставимый].[НДФЛ ],
                            [Measures].[<%fact_measure%>]
                        ) +
                        (
                            [КД].[Сопоставимый].[НДФЛ ],
                            [Measures].[Факт_за период ]
                        ),
                        IIF
                        (
                            (Measures.[Нп] + 30) = 0,
                            null,
                            (Measures.[Нт] + 30)
                            * ( [Measures].[<%fact_measure%>] ) / (Measures.[Нп] + 30)
                        )
                    )
                )
            ),
            [Measures].[<%fact_measure%>]
        )'
    member [Measures].[Темп роста ]
    as 'IIF
        (
            IsEmpty
            (
                (
                    ParallelPeriod
                    (
                        [Период].[Период].[Год],
                        1,
                        [Период].[Период].CurrentMember
                    ),
                    [Measures].[Факт_за период ]
                )
            ) or
            (
                ParallelPeriod
                (
                    [Период].[Период].[Год],
                    1,
                    [Период].[Период].CurrentMember
                ),
                [Measures].[Факт_за период ]
            ) = 0,
            null,
            (
                [Measures].[Факт_за период ],
                [Период].[Период].CurrentMember
            ) /
            (
                ParallelPeriod
                (
                    [Период].[Период].[Год],
                    1,
                    [Период].[Период].CurrentMember
                ),
                [Measures].[Факт_за период ]
            )
        )'
    set [Города]
    as 'Filter
        (
            Descendants
            (
                [Районы].[Сопоставимый].[Все районы],
                <%regions_level%>,
                SELF
            ),
            not ([Районы].[Сопоставимый].CurrentMember is [Районы].[Сопоставимый].CurrentMember.Parent.DataMember ) and
            (
               [Районы].[Сопоставимый].CurrentMember.Properties("Тип территории") = "ГО"
            )
        )'
    set [Районы]
    as 'Filter
        (
            Descendants
            (
                [Районы].[Сопоставимый].[Все районы],
                <%regions_level%>,
                SELF
            ),
            not ([Районы].[Сопоставимый].CurrentMember is [Районы].[Сопоставимый].CurrentMember.Parent.DataMember ) and
            (
                [Районы].[Сопоставимый].CurrentMember.Properties("Тип территории") = "МР"
            )
        )'
        
    set [Города и районы]
    as '[Города] + [Районы]'
    
    member [Районы].[Сопоставимый].[Муниципальные районы ]
    as 'Aggregate({[Районы]})'
    
    member [Районы].[Сопоставимый].[Городские округа ]
    as 'Aggregate({[Города]})'
    
    member Measures.[isMR]
    as 'not
       (
         [Районы].[Сопоставимый].CurrentMember is [Районы].[Сопоставимый].[Консолидированный бюджет субъекта ] or
         [Районы].[Сопоставимый].CurrentMember is [Районы].[Сопоставимый].[Бюджет субъекта] or
         [Районы].[Сопоставимый].CurrentMember is [Районы].[Сопоставимый].[Местные бюджеты]
       )'          
SELECT
    {
        [Measures].[Год],
        [Measures].[Факт_за период ]
    } on columns,
    { [Периоды] } on rows

        <%region_document_skif_type%>
