            [МФ РФ].[Сопост Показатели].[Все показатели].[Федеральный фонд софинансирования расходов]
        )'
    member Measures.[ФФСР] as 'CoalesceEmpty(Measures.[Empty ФФСР], 0)'

        (
            [Measures].[Прогноз_очередной год],
            [МФ РФ].[Сопост Показатели].[Все показатели].[Иные межбюджетные трансферты]

        Measures.[ФФК],
        Measures.[ФФСР],
        Measures.[Иные]
    } on columns,
    {
        {
            [Территории].[Сопоставимый].[Все территории].[Российская  Федерация].[<%region%>].[<%state_area%>]
        } +
        Filter({
            [Территории].[Сопоставимый].[Все территории].[Российская  Федерация].[<%region%>].[<%state_area%>].Siblings
        }  -
        {
            [Территории].[Сопоставимый].[Все территории].[Российская  Федерация].[<%region%>].[<%state_area%>].Parent.DataMember
        }, (Measures.[ФФФПР]+
        Measures.[ФФК]+
        Measures.[ФФСР]) <> 0)
    } on rows
from [МФ РФ_Фонды]
where
    (
        [Период].[Период].[Данные всех периодов].[<%period_year%>],
       [Вариант].[МФРФ Фонды].[Все].[Данные, утвержденные Федеральным законом "О федеральном бюджете"].[ФЗ "О федеральном бюджете на 2011 год и на плановый период 2012 и 2013 годов"].[ФЗ от 13 декабря 2010 г. № 357-ФЗ]
