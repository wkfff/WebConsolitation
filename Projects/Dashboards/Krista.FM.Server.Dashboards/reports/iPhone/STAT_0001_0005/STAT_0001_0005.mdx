            ),
            null,

                [Территории__РФ].[Территории__РФ].CurrentMember,

                    Filter
                    (
                        [Территории__РФ].[Территории__РФ].[Все территории].[Российская Федерация].[Сибирский федеральный округ].[Новосибирская обл.].[Новосибирская область].Children,

                        )  and not IsEmpty
                        (
                            [Measures].[Уровень безработицы]
                        )
                    ),
                    [Measures].[Уровень безработицы],
                    BASC
                ),
                [Measures].[Уровень безработицы]
            )
        )'
    set [Все районы]
    as ' Filter
        (
            [Территории__РФ].[Территории__РФ].[Все территории].[Российская Федерация].[Сибирский федеральный округ].[Новосибирская обл.].[Новосибирская область].Children,

            ) and not IsEmpty
            (
                [Measures].[Уровень безработицы]
            )
        )'
    member [Measures].[Худший ранг ]
    as 'max
        (
            [Все районы],
            [Measures].[Ранг ]
        )'
    member [Measures].[Краткое наименование ]
