with        
    member [ФСТ].[Показатели].[Количество регулируемых организаций электроснабжения, в т.ч.]        
    as '[ФСТ].[Показатели].[Все показатели].[Количество регулируемых организаций электроснабжения, всего]'        
    member [ФСТ].[Показатели].[Региональная генерация]        
    as '[ФСТ].[Показатели].[Все показатели].[Количество регулируемых организаций региональной генерации]'        
    member [ФСТ].[Показатели].[Энергоснабжающие] 
    as '[ФСТ].[Показатели].[Все показатели].[Количество регулируемых энергоснабжающих организации]' 
    member [ФСТ].[Показатели].[Электросетевые] 
    as '[ФСТ].[Показатели].[Все показатели].[Количество регулируемых организаций обслуживающих сети]' 
    member [ФСТ].[Показатели].[Энергосбытовые] 
    as '[ФСТ].[Показатели].[Все показатели].[Количество регулируемых сбытовых организаций]'        
    member [Measures].[Факт прошлый год]       
    as '      
        (
            ParallelPeriod      
            (
                [Период].[Год Квартал Месяц].[Год],
                1,
                [Период].[Год Квартал Месяц].CurrentMember       
            ),
            [Measures].[Факт]     
        ) '      
    member [Measures].[Темп роста]     
    as ' IIF     
        (
            not isEmpty([Measures].[Факт прошлый год]) and not     
            (
                [Measures].[Факт прошлый год] = 0    
            ),
            [Measures].[Факт] - [Measures].[Факт прошлый год]  ,
            null   
        ) '     
select
    {
        {            
            [Период].[Год Квартал Месяц].[Данные всех периодов].[2010],
            [Период].[Год Квартал Месяц].[Данные всех периодов].[2011]
        }     
        *        
        {
            [Measures].[Факт],
            [Measures].[Темп роста]   
        }        
    } on columns,
    {
        [ФСТ].[Показатели].[Количество регулируемых организаций электроснабжения, в т.ч.],
        [ФСТ].[Показатели].[Региональная генерация],
        [ФСТ].[Показатели].[Энергоснабжающие],
        [ФСТ].[Показатели].[Электросетевые],
        [ФСТ].[Показатели].[Энергосбытовые],
        [ФСТ].[Показатели].[Все показатели].[Количество регулируемых организаций теплоснабжения],
        [ФСТ].[Показатели].[Все показатели].[Количество регулируемых организаций водоснабжения]        
    } on rows        
from  [ФСТ_ЖКХ_Тарифы]        
WHERE        
    (
        [Территории].[Сопоставимый].[Все территории].[Российская  Федерация].[<%region%>].[<%state_area%>]        
    )
