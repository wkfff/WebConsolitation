-- FST_0001_0001_Chart
with  
    member [ФСТ].[Показатели].[Среднеотпускной тариф на электрическую энергию ]  
    as 'IIF  
        (
            [ФСТ].[Показатели].[Все показатели].[Среднеотпускной тариф на электрическую энергию] > 0,
            [ФСТ].[Показатели].[Все показатели].[Среднеотпускной тариф на электрическую энергию],
            null
        )'  
    member [ФСТ].[Показатели].[Тариф на электрическую энергию для населения ]  
    as 'IIF  
        (
            [ФСТ].[Показатели].[Все показатели].[Тариф на электрическую энергию для населения] > 0,
            [ФСТ].[Показатели].[Все показатели].[Тариф на электрическую энергию для населения],
            null 
        )'  
select
    {        
        [ФСТ].[Показатели].[Тариф на электрическую энергию для населения ] 
    }on columns,
    non empty
    {
        Descendants  
        (
            [Период].[Год Квартал Месяц].[Данные всех периодов],
            [Период].[Год Квартал Месяц].[Год],
            self  
        ) 
    } on rows   
from  [ФСТ_ЖКХ_Тарифы]
WHERE ( [Measures].[Факт],[Территории].[Сопоставимый].[Все территории].[Российская  Федерация].[<%region%>].[<%state_area%>] ) 
]]>
    </query>

-- FST_0001_0013_gauge1
with  
    member [ФСТ].[Показатели].[Уровень]  
    as '[Территории].[Сопоставимый].CurrentMember.Level.Name'  
    set [Все регионы]  
    as 'filter  
        (
            Descendants  
            (
                [Территории].[Сопоставимый].[Все территории],
                [Территории].[Сопоставимый].[РФ],
                SELF_AND_AFTER  
            ) ,
            not  
            (
                [Территории].[Сопоставимый].CurrentMember is [Территории].[Сопоставимый].CurrentMember.Parent.DataMember  
            ) and not [Территории].[Сопоставимый].CurrentMember is [Территории].[Сопоставимый].[Все территории].[Несопоставленные данные]    and  
            (
                [Территории].[Сопоставимый].CurrentMember.Level.Name = "Субъект РФ"  
            )  
        )'   
    member [ФСТ].[Показатели].[Единицы измерения ]  
    as '"<%measure%>"'   
    member [ФСТ].[Показатели].[Рост тарифа на электрическую энергию ] 
    as '[ФСТ].[Показатели].[Все показатели].[Рост тарифа на электрическую энергию] - 1' 
    member [ФСТ].[Показатели].[Рост тарифа на электрическую энергию для населения ]
    as '[ФСТ].[Показатели].[Все показатели].[Рост тарифа на электрическую энергию для населения] - 1'
select
    {
        [ФСТ].[Показатели].[Все показатели].[Среднеотпускной тариф на электрическую энергию],
        [ФСТ].[Показатели].[Рост тарифа на электрическую энергию ]
    }on columns,
    {
        order  
        (
            filter  
            (          
                [Все регионы],
                not isEmpty  
                (
                    [ФСТ].[Показатели].[Все показатели].[Среднеотпускной тариф на электрическую энергию]
                )  and not [ФСТ].[Показатели].[Все показатели].[Среднеотпускной тариф на электрическую энергию] = 0
            ),
            [ФСТ].[Показатели].[Все показатели].[Среднеотпускной тариф на электрическую энергию],
            BASC  
        )  
    } on rows  
from  [ФСТ_ЖКХ_Тарифы]  
WHERE  
    (
        [Measures].[Факт],
        [Период].[Год Квартал Месяц].[Данные всех периодов].[2011]  
    )
]]>
    </query>
