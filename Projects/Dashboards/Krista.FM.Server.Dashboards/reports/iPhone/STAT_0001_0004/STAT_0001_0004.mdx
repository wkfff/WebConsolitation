with     
    member [Показатели__Рынок труда].[Показатели__Рынок труда].[Уровень безработицы]     
    as 'IIF     
        (
            not isEmpty     
            (
                (
                    [Показатели__Рынок труда].[Показатели__Рынок труда].[Все].[Численность экономически активного населения]     
                )     
            ) and not     
            (
                [Показатели__Рынок труда].[Показатели__Рынок труда].[Все].[Численность экономически активного населения]     
            ) = 0,
            (
                [Показатели__Рынок труда].[Показатели__Рынок труда].[Все].[Численность безработных граждан]     
            )/     
            (
                [Показатели__Рынок труда].[Показатели__Рынок труда].[Все].[Численность экономически активного населения]     
            )              ,
            null     
        )'     
    member [Measures].[Значение ]     
    as '     
        (
            [Measures].[Значение],
            <%period_current_date%>
        )' 
    member [Measures].[Значение по Югре] 
    as ' 
        (
            [Measures].[Значение],
            <%period_current_date%>,
            [Территории__РФ].[Территории__РФ].[Все территории].[Российская Федерация].[Уральский федеральный округ].[Тюменская область с Ханты-Мансийским автономным округом, Ямало-Ненецким автономным округом].[Ханты-Мансийский автономный округ - Югра] 
        )' 
    member [Measures].[Разница с югрой] 
    as ' [Measures].[Значение ] - [Measures].[Значение по Югре]' 
    member Measures.[Численность экономически активного населения ]     
    as '     
        (
            [Measures].[Значение],
            [Показатели__Рынок труда].[Показатели__Рынок труда].[Все].[Численность экономически активного населения]     
        )'     
    member [Measures].[Ранг ]
    as '[Measures].[Обратный ранг]'
    member [Measures].[Обратный ранг]
    as 'iif(IsEmpty([Measures].[Прямой ранг]), null, Measures.[Худший ранг ] + 1 - [Measures].[Прямой ранг])'
    member Measures.[Прямой ранг]              
    as 'IIF     
        (
            IsEmpty (  [Measures].[Значение]  )  or   
            (
                [Территории__РФ].[Территории__РФ].CurrentMember is [Территории__РФ].[Территории__РФ].[Все территории].[Российская Федерация].[Уральский федеральный округ].[Тюменская область с Ханты-Мансийским автономным округом, Ямало-Ненецким автономным округом].[Ханты-Мансийский автономный округ - Югра]   
            ),
            null,
            Rank     
            (
                [Территории__РФ].[Территории__РФ].CurrentMember,
                Order     
                (
