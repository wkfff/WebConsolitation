-- FO_0003_0009_date
with
    member Measures.[ДанныеНа]
    as '[Период].[Период].CurrentMember.UniqueName'
select
    {Measures.[ДанныеНа]} on columns,
    {
        Tail
        (
            NonEmptyCrossjoin
            (
                {
                    Filter
                    (
                        [Период].[Период].Members,
                        [Период].[Период].CurrentMember.Name <> "Остатки на начало года" and
                        [Период].[Период].CurrentMember.Name <> "Данные месяца" and
                        [Период].[Период].CurrentMember.Name <> "Данные квартала" and
                        [Период].[Период].CurrentMember.Name <> "Данные полугодия" and
                        [Период].[Период].CurrentMember.Name <> "Данные года"
                    )
                }
            )
        )
    } on rows
from  [ФК_Паспорт субъекта]
  ]]>
  </query>

 

    <query id="FO_0003_0009_chart">
        <![CDATA[
-- FO_0003_0009_chart
with 
    member [Показатели].[Паспорт субъекта].[Раздел I. Социально-значимые расходы ]
    as 'IIF 
        (
            not isEmpty 
            (
                [Показатели].[Паспорт субъекта].[Всего] 
            ) and not 
            (
                [Показатели].[Паспорт субъекта].[Всего] = 0 
            ),
            [Показатели].[Паспорт субъекта].[Все].[Раздел I. Социально-значимые расходы] / [Показатели].[Паспорт субъекта].[Всего] 
            * 100,
            0 
        )' 
    member [Показатели].[Паспорт субъекта].[Раздел II. Первоочередные расходы ]
    as 'IIF 
        (
            not isEmpty 
            (
                [Показатели].[Паспорт субъекта].[Всего] 
            ) and not 
            (
                [Показатели].[Паспорт субъекта].[Всего] = 0 
            ),
            [Показатели].[Паспорт субъекта].[Все].[Раздел II. Первоочередные расходы] / [Показатели].[Паспорт субъекта].[Всего] 
            * 100,
            0 
        )' 
    member [Показатели].[Паспорт субъекта].[Раздел III.Расходы ]
    as 'IIF 
        (
            not isEmpty 
            (
                [Показатели].[Паспорт субъекта].[Всего] 
            ) and not 
            (
                [Показатели].[Паспорт субъекта].[Всего] = 0 
            ),
            [Показатели].[Паспорт субъекта].[Все].[Раздел III.Расходы] / [Показатели].[Паспорт субъекта].[Всего] 
            * 100,
            0 
        )' 
    member [Показатели].[Паспорт субъекта].[Всего] 
    as 'sum 
        (
            {
                [Показатели].[Паспорт субъекта].[Все].[Раздел I. Социально-значимые расходы],
                [Показатели].[Паспорт субъекта].[Все].[Раздел II. Первоочередные расходы],
                [Показатели].[Паспорт субъекта].[Все].[Раздел III.Расходы] 
            } 
        )' 
    member [Measures].[Исполнение консолидированного бюджета ] 
    as '[Measures].[Исполнение консолидированного бюджета]' 
    member [Период].[Период].[План ] 
    as ' 
        (
            [Период].[Период].[Данные всех периодов].[<%period_year%>],
            [Measures].[Первоначальный план] 
        )' 
    member [Период].[Период].[Уточненный план ] 
    as ' 
        (
            <%period_cur_date%>,
            [Measures].[Уточненный план консолидированного бюджета] 
        )' 
    member [Период].[Период].[Текущее исполнение] 
    as ' 
        (
            <%period_cur_date%>,
            [Measures].[Исполнение консолидированного бюджета ] 
        )' 
    member [Период].[Период].[Прошлый год] 
    as ' 
        (
            [Период].[Период].[Данные всех периодов].[<%period_last_year%>],
            [Measures].[Исполнение консолидированного бюджета ] 
        )' 
    member [Период].[Период].[2 года назад] 
    as ' 
        (
            [Период].[Период].[Данные всех периодов].[<%period_last_last_year%>],
            [Measures].[Исполнение консолидированного бюджета ] 
        )' 
select
    {
        [Период].[Период].[План ],
        [Период].[Период].[Уточненный план ],
        [Период].[Период].[Текущее исполнение] ,
        [Период].[Период].[Прошлый год],
        [Период].[Период].[2 года назад] 
    }on columns,
    {
        [Показатели].[Паспорт субъекта].[Раздел I. Социально-значимые расходы ],
        [Показатели].[Паспорт субъекта].[Раздел II. Первоочередные расходы ],
        [Показатели].[Паспорт субъекта].[Раздел III.Расходы ]
    } on rows 
from  [ФК_Паспорт субъекта] 
where       
    (
        [Measures].[Первоначальный план],
        [Территории].[Сопоставимый].[Все территории].[Российская  Федерация].[<%region%>].[<%state_area%>]         
    )  
    ]]>  
    </query>

            [Measures].[Исполнение консолидированного бюджета ]

    }  on rows

    (
        [Территории].[Сопоставимый].[Все территории].[Российская  Федерация].[<%region%>].[<%state_area%>]  
    )
