    member [Территории].[РФ].[Все территории].[Российская Федерация].[Уральский федеральный округ ]
    as '[Территории].[РФ].[Все территории].[Российская Федерация].[Уральский федеральный округ].DataMember'
    set [Список все субъектов УФО]
    as 'Filter
        (
            [Территории].[РФ].[Все территории].[Российская Федерация].[Уральский федеральный округ].Children,
            not
            (
                [Территории].[РФ].CurrentMember is [Территории].[РФ].CurrentMember.Parent.DataMember
            )
        ) +
        {
            [Территории].[РФ].[Все территории].[Российская Федерация].[Уральский федеральный округ ]
        }'
    member [Труд].[Трудовые ресурсы].[Уровень общей безработицы по методологии МОТ ]
    as '
        (
            [Труд].[Трудовые ресурсы].[Все показатели].[Уровень общей безработицы (от экономически активного населения)],
            [Территории].[РФ].CurrentMember
        ) * 100 '
    member    [Труд].[Трудовые ресурсы].[Уровень общей безработицы по методологии МОТ РФ ]
    as '
        (
            [Труд].[Трудовые ресурсы].[Все показатели].[Уровень общей безработицы (от экономически активного населения)],
            [Территории].[РФ].[Все территории].[Российская Федерация].DATAMEMBER ,
            [Источники данных].[Все источники данных].[СТАТ Отчетность - ФСГС]
        ) '
    member [Период].[Год Квартал Месяц].[Предыдущий месяц] 
    as 'IIF 
        (
            <%period_year%>.PrevMember is 
            <%period_year%>.PrevMember.Parent.DataMember,
            <%period_year%>.PrevMember.PrevMember,
            <%period_year%>.PrevMember 
        )'
    member [Труд].[Трудовые ресурсы].[Численность экономически активного населения ]
    as '[Труд].[Трудовые ресурсы].[Все показатели].[Численность экономически активного населения]'
    member [Труд].[Трудовые ресурсы].[Общая численность безработных по методологии МОТ ]
    as '[Труд].[Трудовые ресурсы].[Все показатели].[Общая численность безработных (на конец месяца)]'
    member measures.[Значение ]
    as '
        (
            [Труд].[Трудовые ресурсы].CurrentMember,
            <%period_year%>,
            [Measures].[Нарастающий итог]
        )'
    member measures.[Значение прошлый месяц ]
    as '
        (
            [Труд].[Трудовые ресурсы].CurrentMember,
            [Период].[Год Квартал Месяц].[Предыдущий месяц] ,
            [Measures].[Нарастающий итог]
        )'
    member measures.[Темп прироста ]
    as 'measures.[Значение ] / measures.[Значение прошлый месяц ] - 1'
    Member [Measures].[Прирост ]
    as 'IIF
        (
            ([Труд].[Трудовые ресурсы].CurrentMember is [Труд].[Трудовые ресурсы].[Уровень общей безработицы по методологии МОТ ]),
            Rank
            (
                [Территории].[РФ].CurrentMember,
                Order
                (
                    Filter
                    (
                        [Территории].[РФ].CurrentMember.Siblings,
                        not IsEmpty
                        (
                            (
                                measures.[Значение ],
                                [Труд].[Трудовые ресурсы].[Уровень общей безработицы по методологии МОТ ]
                            )
                        )
                    ),
                    (
                        measures.[Значение ],
                        [Труд].[Трудовые ресурсы].[Уровень общей безработицы по методологии МОТ ]
                    ) ,
                    DESC
                ),
                (
                     measures.[Значение ],
                     [Труд].[Трудовые ресурсы].[Уровень общей безработицы по методологии МОТ ]
                )
            ) ,
            measures.[Значение ] - measures.[Значение прошлый месяц ]
        )'
SELECT
    { [Список все субъектов УФО] } ON COLUMNS ,
    {
        {
            [Труд].[Трудовые ресурсы].[Численность экономически активного населения ],
            [Труд].[Трудовые ресурсы].[Общая численность безработных по методологии МОТ ],
            [Труд].[Трудовые ресурсы].[Уровень общей безработицы по методологии МОТ ]
        }
        *
        {
            measures.[Значение ],
            measures.[Темп прироста ],
            measures.[Прирост ]
        }
    } ON ROWS
FROM [СТАТ_Труд_Трудовые ресурсы]
WHERE
    (
        [Группировки].[Труд_Трудовые ресурсы].[Все группировки].[Без группировки] ,
        [Источники данных].[Все источники данных].[СТАТ Отчетность - ФСГС по Свердловской области]
    )
  ]]>
</query>
