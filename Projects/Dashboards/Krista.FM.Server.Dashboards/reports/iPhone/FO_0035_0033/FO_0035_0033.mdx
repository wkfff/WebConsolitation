    as '[Measures].[Перечислено] / 1000'

    as 'filter  
        (
            [Администратор__ИспКасПланСопост].[Администратор__ИспКасПланСопост].[ГРБС].members,
            not  
            (
                [Администратор__ИспКасПланСопост].[Администратор__ИспКасПланСопост].CurrentMember is [Администратор__ИспКасПланСопост].[Администратор__ИспКасПланСопост].[Все].[Несопоставленные данные] 
            )  
        )'  
    member [Администратор__ИспКасПланСопост].[Администратор__ИспКасПланСопост].[Всего]  
    as 'Aggregate ( [ГРБС] )'  
select
    {        
        [Measures].[Поступило из ФБ ],
        [Measures].[Перечислено ],
        [Measures].[Остаток ] 
    } on columns,
    {
        [Администратор__ИспКасПланСопост].[Администратор__ИспКасПланСопост].[Всего],
        [ГРБС]  
    }   on rows

    (
        <%selected_period%> 
    )  
  ]]>
