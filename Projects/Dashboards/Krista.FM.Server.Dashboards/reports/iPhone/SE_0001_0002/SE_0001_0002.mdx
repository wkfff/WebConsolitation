  <query id="SE_0001_0002_loaded">
    <![CDATA[
    --SE_0001_0002_loaded
with
    member [Measures].[Отгружено ]
    as '
        (
            [Measures].[За период],
            [Период].[Год Квартал Месяц].[Данные всех периодов]<%period_month%>
        )'
    member [Measures].[Отгружено прошлый год]
    as '
        (
            [Measures].[За период],
            [Период].[Год Квартал Месяц].[Данные всех периодов]<%period_last_date%>
        )'
    member [Measures].[Прирост ]
    as 'IIF
        (
            not isEmpty
            (
                [Measures].[Отгружено прошлый год]
            ) and not
            (
                [Measures].[Отгружено прошлый год]  = 0
            ) ,
            [Measures].[Отгружено ] / [Measures].[Отгружено прошлый год],
            null
        ) - 1'
        set [Территории ]
    as 'Filter
        (
            [Территории].[СЭП].Children,
            not IsEmpty ( [Measures].[Отгружено ] ) and
            (
                [Территории].[СЭП].CurrentMember.Properties("Принадлежность") = [Территории].[СЭП].[Все территории].[<%state_area%>].Properties("Принадлежность")
            )
        )'
    set [Территории РФ]
    as 'Filter
        (
            [Территории].[СЭП].Children,
            not IsEmpty ( [Measures].[Отгружено ] ) and not
            (
                [Территории].[СЭП].CurrentMember.Properties("Принадлежность") = "РФ"
            ) and not
            (
                [Территории].[СЭП].CurrentMember is [Территории].[СЭП].[Все территории].[Российская Федерация]
            )
        )'
    member measures.[Ранг по РФ]
    as ' Rank
        (
            [Территории].[СЭП].CurrentMember,
            Order
            (
                [Территории РФ],
                ( [Measures].[Отгружено ] ),
                DESC
            ),
            ( [Measures].[Отгружено ] )
        ) '
    member Measures.[Ранг по ФО]
    as ' Rank
        (
            [Территории].[СЭП].CurrentMember,
            Order
            (
                [Территории ],
                ( [Measures].[Отгружено ] ),
                BDESC
            ),
            ( [Measures].[За период] )
        ) '
    member measures.[Худший ранг ФО]
    as ' max
        (
            [Территории ],
            Measures.[Ранг по ФО]
        ) '
    member measures.[Худший ранг РФ]
    as ' max
        (
            [Территории РФ],
            Measures.[Ранг по РФ]
        ) '
member [Measures].[Доля ФО]
    as 'IIF 
        (
            not isEmpty 
            (
                (
                    [Measures].[Отгружено ],
                    [Территории].[СЭП].[Все территории].[<%region%>]
                ) 
            ) and not 
            (
                (
                    [Measures].[Отгружено ],
                    [Территории].[СЭП].[Все территории].[<%region%>]
                )  = 0 
            ) ,
            [Measures].[Отгружено ] / 
            (
                [Measures].[Отгружено ],
                [Территории].[СЭП].[Все территории].[<%region%>]
            ),
            null 
        )' 
        member [Measures].[Доля РФ]
    as 'IIF
        (
            not isEmpty
            (
                (
                    [Measures].[Отгружено ],
                    [Территории].[СЭП].[Все территории].[Российская Федерация]
                )
            ) and not
            (
                (
                    [Measures].[Отгружено ],
                    [Территории].[СЭП].[Все территории].[Российская Федерация]
                )  = 0
            ) ,
            [Measures].[Отгружено ] /
            (
                [Measures].[Отгружено ],
                [Территории].[СЭП].[Все территории].[Российская Федерация]
            ),
            null
        )'
        select
    {
        [Measures].[Отгружено ],
        [Measures].[Отгружено прошлый год] ,
        [Measures].[Прирост ] ,
        measures.[Ранг по РФ],
        measures.[Худший ранг РФ],
        Measures.[Ранг по ФО],
        measures.[Худший ранг ФО],
        [Measures].[Доля ФО],
        [Measures].[Доля РФ]
    } on columns,
    {
        [Территории].[СЭП].[Все территории].[<%state_area%>],
        [Территории].[СЭП].[Все территории].[Российская Федерация],
