-- FO_0035_0020_date

                        [Период].[Период].CurrentMember.Name <> "Данные года" 
                    ) 
                } 
            ) 
        )

    <![CDATA[
-- FO_0035_0021_data
    with 
    member [Measures].[Первоначальный кассовый план ]
    as '[Measures].[Первоначальный кассовый план] / 1000'
    member [Measures].[Уточненный кассовый план ]
    as '[Measures].[Уточненный кассовый план] / 1000'
    member [Measures].[Исполнено с начала года ]
    as '([Measures].[Исполнено с начала года] + [Measures].[Исполнено с начала месяца])/ 1000'   
    
    member [Measures].[Процент исполнения ]
    as 'IIF
        (
           IsEmpty([Measures].[Уточненный кассовый план ])  or ([Measures].[Уточненный кассовый план ]) = 0,
           null,
           [Measures].[Исполнено с начала года ] / [Measures].[Уточненный кассовый план ]
        )'

    select
    {
        [Measures].[Первоначальный кассовый план ],
        [Measures].[Уточненный кассовый план ],
        [Measures].[Исполнено с начала года ],
        [Measures].[Процент исполнения ] 
    }    on columns,
    {
        [Показатели].[Ожидаемое исполнение бюджета].[Все].[Доходы - всего и ИФДБ],
        [Показатели].[Ожидаемое исполнение бюджета].[Все].[Расходы - всего],
        [Показатели].[Ожидаемое исполнение бюджета].[Все].[Отклонение доходов от расходов]  
    } on rows   
from  [ФО_Ожидаемое исполнение бюджета]
where   
    (
        <%period_cur_date%> 
    )  
  ]]>
