   as 'IIF ([Показатели].[ФО_Исполнение бюджета_Сопоставимый].CurrentMember is [Показатели].[ФО_Исполнение бюджета_Сопоставимый].[Все].[НАЛОГОВЫЕ И НЕНАЛОГОВЫЕ ДОХОДЫ] or

              ([Measures].[Исполнено на дату] / [Measures].[Исполнено на дату_Прошлый год])*100 - 100

              [Measures].[Годовые назначения] = 0,
              null,
