-- FO_0035_0034_ReviewGrid
with   
    member [Measures].[Поступило из ФБ ]   
    as '[Measures].[Поступило из ФБ] / 1000'   
    member [Measures].[Перечислено ]   
    as '[Measures].[Перечислено] / 1000'   
    member [Measures].[Остаток ]   
    as '[Measures].[Остаток] / 1000'   
    member [Measures].[Код] 
    as '[Код цели__Сопоставимый].[Код цели__Сопоставимый].CurrentMember.Properties("Код")' 
    member [Код цели__Сопоставимый].[Код цели__Сопоставимый].[Всего]  
    as 'Aggregate ( [Цели] )'  
    set [Цели]  
    as 'filter  
        (
            Descendants  
            (
                [Код цели__Сопоставимый].[Код цели__Сопоставимый].[Все],
                [Код цели__Сопоставимый].[Код цели__Сопоставимый].[Код цели],
                SELF  
            ),
            not  [Код цели__Сопоставимый].[Код цели__Сопоставимый].CurrentMember is [Код цели__Сопоставимый].[Код цели__Сопоставимый].CurrentMember.Parent.Datamember and  
            (
                not isEmpty ( [Measures].[Поступило из ФБ ]) or not isEmpty ( [Measures].[Перечислено ] ) or not isEmpty ( [Measures].[Остаток ] )  
            )  
        )'  
select
    {
        [Measures].[Код],
        [Measures].[Поступило из ФБ ],
        [Measures].[Перечислено ],
        [Measures].[Остаток ]   
    } on columns,
    {
        [Код цели__Сопоставимый].[Код цели__Сопоставимый].[Всего],
        [Цели]  
    } ON rows  
from   [ФО_Остатки средств бюджета_ГРБС] 
WHERE       
    (        
        <%selected_period%>,    
        [Администратор__ИспКасПланСопост].[Администратор__ИспКасПланСопост].[Все].[<%grbs%>]
    )      
  ]]>
</query>
