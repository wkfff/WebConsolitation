--STAT_0003_0009 Grid  
with member [Measures].[Прирост цепной1]
as '
iif
(
(
                    ParallelPeriod
                    (
                        [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].[Год],
                        1,
                        [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].CurrentMember
                    ),
                    [Measures].[За период]

            )<>0,
            [Measures].[Прирост цепной],
            null
)
'

    as '[Здравоохранение__Заболеваемость населения].[Здравоохранение__Заболеваемость населения].Currentmember.Properties("Единица измерения")'  
select
    non empty  
    {
        {
            filter
           (
                {
                    [Здравоохранение__Заболеваемость населения].[Здравоохранение__Заболеваемость населения].[Все показатели].[Число проведенных операций]  ,
                    [Здравоохранение__Заболеваемость населения].[Здравоохранение__Заболеваемость населения].[Все показатели].[Число проведенных операций с применением высоких медицинских технологий],

                    [Здравоохранение__Заболеваемость населения].[Здравоохранение__Заболеваемость населения].[Все показатели].[Умерло после проведенных операций с применением высоких медицинских технологий]  
                }
                *  
                {
                [Measures].[Ед],
                    [Measures].[За период],
                    [Measures].[Прирост цепной1],
                    [Measures].[Темп прироста цепной] 
                } ,
                not isempty( [Measures].[За период])
            )
        }   
    }    on rows,
    {

        filter 
        (
            {
                [Здравоохранение__Виды операций].members  
            }   ,
            not isempty( [Measures].[За период])  
        )  
    } on columns      
from   [СТАТ_Здравоохранение_Заболеваемость населения]    
where    
    (
        [Источники данных].[Источники данных].[Источник].[СТАТ Отчетность - Статистический раздел] ,
    [ОК__Отдельные возрастные группы].[Все отдельные группы].[Значение не указано] ,
       [ОК__МКБ10].[Все виды заболеваний].[Значение не указано],
        [Группировки__Здравоохранение_Заболеваемость].[Все группировки].[По городам и районам, по видам операций] ,
        <%baseRegion%>,
        <%selectedDate%>
    )    
   ]]>
  </query>
