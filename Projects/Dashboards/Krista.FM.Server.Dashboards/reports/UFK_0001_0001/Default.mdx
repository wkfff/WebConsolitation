    --UFK_0001_0001_data--
  with
    member Measures.[ДанныеНа]
    as ' [Период__Период].[Период__Период].CurrentMember.UniqueName'

    {Measures.[ДанныеНа]} on columns,
    {
        Tail
        (
            Filter
            (
               [Период__Период].[Период__Период].[Месяц].Members,
                not IsEmpty( [Measures].[Сумма с начала года] )
            ),
            1
        )
    } on rows

    ]]>
