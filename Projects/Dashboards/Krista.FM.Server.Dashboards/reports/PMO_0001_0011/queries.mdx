        [Паспорт МО].[Показатели].[ Жилищный фонд].children + [Паспорт МО].[Показатели].[Ввод в действие жилья, объектов соцкультбыта и здравоохранения].children,

    (
        [Источники данных].[Все источники данных].[МО Паспорт МО],

    )
    ]]>
