    member [Measures].[ДанныеНа]     
    as '[Период].[Период].CurrentMember.UniqueName'

        ) '

    )

<query id="FO_0005_0001_Regions">
<![CDATA[
With   
    member Measures.[Территория]   
    as '[Территории].[Сопоставимый].CurrentMember.Parent.Name' 
SELECT
    { Measures.[Территория] } on columns,
    Filter   
    (
        [Территории].[Сопоставимый].[Все территории].[Российская  Федерация].[Уральский федеральный округ].Children,
        (
            not   
            (
                [Территории].[Сопоставимый].CurrentMember is [Территории].[Сопоставимый].CurrentMember.Parent.DataMember   
            )   
        )   
    )  on rows   
FROM [ФО_МесОтч_СпрЗадолженность]   
  ]]>
</query>

<query id="FO_0005_0001_Grid">
    <![CDATA[
   -- FO_0005_0001_Grid
with               
    Member [Measures].[Факт ]               
    as 'IIF               
        (
            [Measures].[Факт] = 0,
            null,
            [Measures].[Факт]/1000               
        )'               
    member Measures.[Аналогичный месяц прошлого года]               
    as '               
        (
            [Measures].[Факт ],
            [Период].[Период].[Данные всех периодов].[<%period_last_year%>].[<%period_half_year%>].[<%period_quater%>].[<%period_month%>]                          
        )'               
    member Measures.[Декабрь прошлого года]               
    as '               
        (
            [Measures].[Факт ],
            [Период].[Период].[Данные всех периодов].[<%period_last_year%>].[Полугодие 2].[Квартал 4].[Декабрь]               
        )'               
    member Measures.[Прошлый месяц текущего года]               
    as '               
        (
            [Measures].[Факт ],
            [Период].[Период].[Данные всех периодов].[<%period_year%>].[<%lastHalf%>].[<%lastQuater%>].[<%lastMonth%>]               
        )'               
    member Measures.[Текущий месяц текущего года]               
    as '               
        (
            [Measures].[Факт ],
            [Период].[Период].[Данные всех периодов].[<%period_year%>].[<%period_half_year%>].[<%period_quater%>].[<%period_month%>]               
        )'               
    member Measures.[Отклонение от аналогичного периода предыдущего года]               
    as '               
        (
            Measures.[Текущий месяц текущего года] -  Measures.[Аналогичный месяц прошлого года]               
        )'               
    member Measures.[Темп роста к аналогичному периоду]               
    as 'IIF               
        (
            IsEmpty               
            (
                Measures.[Аналогичный месяц прошлого года]               
            ) or               
            (
                Measures.[Аналогичный месяц прошлого года] =0               
            ),
            null,
            (
                Measures.[Текущий месяц текущего года] / Measures.[Аналогичный месяц прошлого года]               
            )               
        )'               
    member Measures.[Изменение с начала года]               
    as '               
        (
            Measures.[Текущий месяц текущего года] - Measures.[Декабрь прошлого года]               
        )'               
    member Measures.[Темп роста к началу года]               
    as 'IIF               
        (
            IsEmpty               
            (
                Measures.[Декабрь прошлого года]               
            ) or               
            (
                Measures.[Декабрь прошлого года] =0               
            ),
            null,
            (
                Measures.[Текущий месяц текущего года] / Measures.[Декабрь прошлого года]               
            )               
        )'               
    member Measures.[Изменение относительно предыдущего месяца]               
    as '               
        (
            Measures.[Текущий месяц текущего года] - Measures.[Прошлый месяц текущего года]               
        )'               
    member Measures.[Темп роста к предыдущему месяцу]               
    as 'IIF               
        (
            IsEmpty               
            (
                Measures.[Прошлый месяц текущего года]               
            ) or               
            (
                Measures.[Прошлый месяц текущего года] = 0               
            ),
            null,
            (
                Measures.[Текущий месяц текущего года] / Measures.[Прошлый месяц текущего года]               
            )               
        )'                
    member [Measures].[Удельный вес в общей величине задолженности]            
    as 'IIF       
        (
            IsEmpty     
            (
                (
                    Measures.[Текущий месяц текущего года],
                    [Показатели].[СопМесОСпЗадо].[Все показатели]      
                )      
            ) or     
            (
                Measures.[Текущий месяц текущего года],
                [Показатели].[СопМесОСпЗадо].[Все показатели]     
            ) = 0,
            null,
            (
                Measures.[Текущий месяц текущего года] /    
                (
                    Measures.[Текущий месяц текущего года],
                    [Показатели].[СопМесОСпЗадо].[Все показатели]    
                )    
            )    
        )'       
    set [Виды задолженности]               
    as 'Filter               
        (
            Descendants               
            (
                [Показатели].[СопМесОСпЗадо].[Все показатели],
                [Показатели].[СопМесОСпЗадо].[Уровень 2],
                self_and_before               
            )  ,
            not               
            (
                [Показатели].[СопМесОСпЗадо].CurrentMember is [Показатели].[СопМесОСпЗадо].CurrentMember.Parent.DataMember              
            ) and not              
            (
                [Показатели].[СопМесОСпЗадо].CurrentMember is [Показатели].[СопМесОСпЗадо].[Все показатели].[Несопоставленные данные]              
            )
            and not
            ( [Показатели].[СопМесОСпЗадо].CurrentMember is [Показатели].[СопМесОСпЗадо].[Все показатели] )
        )'  
    member [Показатели].[СопМесОСпЗадо].[Прочая просроченная кредиторская задолженность]  
    as ' 
        (
            [Показатели].[СопМесОСпЗадо].[Все показатели].[ПРОСРОЧЕННАЯ КРЕДИТОРСКАЯ  ЗАДОЛЖЕННОСТЬ, всего].DATAMEMBER  
        )'  
SELECT
    {
        Measures.[Аналогичный месяц прошлого года],
        Measures.[Декабрь прошлого года],
        Measures.[Прошлый месяц текущего года],
        Measures.[Текущий месяц текущего года],
        Measures.[Отклонение от аналогичного периода предыдущего года],
        Measures.[Темп роста к аналогичному периоду],
        Measures.[Изменение с начала года],
        Measures.[Темп роста к началу года],
        Measures.[Изменение относительно предыдущего месяца],
        Measures.[Темп роста к предыдущему месяцу],
        [Measures].[Удельный вес в общей величине задолженности]        
    }on columns,
   NON EMPTY  {
        [Виды задолженности] + 
        {
            [Показатели].[СопМесОСпЗадо].[Прочая просроченная кредиторская задолженность] 
        }   
    }on rows 
FROM [ФО_МесОтч_СпрЗадолженность]               
Where               
    (
        [ТипДокумента].[СКИФ].[Все].[Консолидированная отчетность и отчетность внебюджетных территориальных фондов],
		[Территории].[Сопоставимый].[Все территории].[Российская  Федерация].[Уральский федеральный округ]<%subject_FO%>,
		<%skif_level%>
    )               
    ]]>
  </query>
  
  
 <query id="FO_0005_0001_Chart">
 <![CDATA[
  --FO_0005_0001_Chart
    with                      
    Member [Measures].[Факт ]                      
    as 'IIF                      
        (
            [Measures].[Факт] = 0,
            null,
            [Measures].[Факт] / 1000                      
        )'                       
    set [Виды задолженности]                      
    as 'Filter                      
        (
            Descendants                      
            (
                [Показатели].[СопМесОСпЗадо].[Все показатели],
                [Показатели].[СопМесОСпЗадо].[Уровень 2],
                self_and_before                      
            )  ,
            not                      
            (
                [Показатели].[СопМесОСпЗадо].CurrentMember is [Показатели].[СопМесОСпЗадо].CurrentMember.Parent.DataMember                     
            ) and not                     
            (
                [Показатели].[СопМесОСпЗадо].CurrentMember is [Показатели].[СопМесОСпЗадо].[Все показатели].[Несопоставленные данные]                     
            ) and not       
            (
                [Показатели].[СопМесОСпЗадо].CurrentMember is [Показатели].[СопМесОСпЗадо].[Все показатели]       
            )        
        )'   
    member [Показатели].[СопМесОСпЗадо].[Прочая просроченная кредиторская задолженность]   
    as '  
        (
            [Показатели].[СопМесОСпЗадо].[Все показатели].[ПРОСРОЧЕННАЯ КРЕДИТОРСКАЯ  ЗАДОЛЖЕННОСТЬ, всего]   
        )'   
    set [Месяцы]       
    as 'Filter     
        (
            descendants      
            (
                [Период].[Период].[Данные всех периодов].[<%period_last_year%>],
                [Период].[Период].[Месяц],
                Self     
            ) + descendants   
            (
                [Период].[Период].[Данные всех периодов].[<%period_year%>],
                [Период].[Период].[Месяц],
                Self   
            ) ,
			not IsEmpty
			(
			    [Measures].[Факт ]   
			)
			and
            (
                [Период].[Период].CurrentMember.Name <> "Остатки на начало года"   
            ) and    
            (
                [Период].[Период].CurrentMember.Name <> "Данные квартала"    
            ) and    
            (
                [Период].[Период].CurrentMember.Name <> "Данные полугодия"    
            ) and    
            (
                [Период].[Период].CurrentMember.Name <> "Данные года"    
            )    
        )'      
SELECT
   NON EMPTY {
        [Виды задолженности] +
        {
            [Показатели].[СопМесОСпЗадо].[Прочая просроченная кредиторская задолженность]
        } 
    }on columns,
    { [Месяцы]   }on rows     
FROM [ФО_МесОтч_СпрЗадолженность]                      
Where                      
    (
        [ТипДокумента].[СКИФ].[Все].[Консолидированная отчетность и отчетность внебюджетных территориальных фондов],
		[Территории].[Сопоставимый].[Все территории].[Российская  Федерация].[Уральский федеральный округ]<%subject_FO%>,
		<%skif_level%>,
        [Measures].[Факт ]       
    )                      
    ]]>
  </query>
