<query id="Grid">
    <![CDATA[
--STAT_0003_0007 Grid1
with member [Measures].[Прирост цепной1]
as '
iif
(
(

                    ParallelPeriod
                    (
                        [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].[Год],
                        1,
                        [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].CurrentMember
                    ),
                    [Measures].[За период]

            )<>0,
            [Measures].[Прирост цепной],
            null

)
'
 member [Measures].[Ед] 
    as ' [Население__Естественное движение].[Население__Естественное движение].Currentmember.Properties("Единица измерения")' 
SELECT
  NON EMPTY
    {
      [Население__Естественное движение].[Население__Естественное движение].[Все показатели].[Коэффициент рождаемости],
      [Население__Естественное движение].[Население__Естественное движение].[Все показатели].[Коэффициент смертности],
      [Население__Естественное движение].[Население__Естественное движение].[Все показатели].[Коэффициент естественного прироста населения],
      [Население__Естественное движение].[Население__Естественное движение].[Все показатели].[Младенческая смертность]--,      [Население__Естественное движение].[Население__Естественное движение].[Все показатели].[Перинатальная смертность]
    } ON COLUMNS
 ,
    {
     {
            [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].[Данные всех периодов].[1998]
        }
        * { [Measures].[Ед]},
        filter 
        (
            {
                [Период__Год Квартал Месяц].[Период__Год Квартал Месяц].[Год].members
            } 
            *
            {
                [Measures].[За период],
                [Measures].[Прирост цепной1],
                [Measures].[Темп прироста цепной]
            },
            not isempty( [Measures].[За период]) 
            and  (
               not isempty( [Население__Естественное движение].[Население__Естественное движение].[Все показатели].[Коэффициент рождаемости])
               
               or
               not isempty( [Население__Естественное движение].[Население__Естественное движение].[Все показатели].[Коэффициент смертности])
               or
               not isempty( [Население__Естественное движение].[Население__Естественное движение].[Все показатели].[Коэффициент естественного прироста населения])
               or
               not isempty( [Население__Естественное движение].[Население__Естественное движение].[Все показатели].[Младенческая смертность])
               )
        ) 


    } ON ROWS
FROM [СТАТ_Население_Естественное движение]
WHERE
  (
  --[Источники данных].[Все источники данных].[СТАТ Отчетность - ДепЗдрав ХМАО-Югры],
   [Население__Очередность рождения].[Все рождения]
   ,[ОК__Основные возрастные группы].[Все возрастные группы].[Значение не указано]
   ,[ОК__Отдельные возрастные группы].[Все отдельные группы].[Значение не указано]
   ,[Половая принадлежность].[Все типы].[Значение не указано]
   ,<%baseRegion%>
   ,[Тип местности].[Все типы местности].[Значение не указано],
   <%typeGroup%>
  )
   ]]>
