    <![CDATA[
-- STAT_0001_0008_grid_activityKinds
WITH 
member measures.[Нарастающий итог ]
as '[Measures].[Нарастающий итог] / 1000000'
    SET [NonFilterColumnSet] 
    AS 'VisualTotals 
        (
            Intersect 
            (
                {
                    {
                        Descendants 
                        (
                            [Период].[Год Квартал Месяц].[Данные всех периодов],
                            [Период].[Год Квартал Месяц].[Данные всех периодов].Level,
                            SELF_AND_AFTER 
                        ) 
                    } 
                },
                {
                    [Период].[Год Квартал Месяц].[Квартал].AllMembers 
                } 
            ) 
        ) 
        * 
        {
            [Measures].[Нарастающий итог ],
            [Measures].[Темп роста к АППГ]
        }' 
    member [Measures].[Темп роста к АППГ]
    as 'IIF
        (
            IsEmpty
            (
                (
                    ParallelPeriod
                    (
                        [Период].[Год Квартал Месяц].[Год],
                        1,
                        [Период].[Год Квартал Месяц].CurrentMember
                    ),
                    [Measures].[Нарастающий итог ]
                )
            ) or
            (
                ParallelPeriod
                (
                    [Период].[Год Квартал Месяц].[Год],
                    1,
                    [Период].[Год Квартал Месяц].CurrentMember
                ),
                [Measures].[Нарастающий итог ]
            ) = 0,
            null,
            (
                [Период].[Год Квартал Месяц].CurrentMember,
                [Measures].[Нарастающий итог ]
            ) /
            (
                ParallelPeriod
                (
                    [Период].[Год Квартал Месяц].[Год],
                    1,
                    [Период].[Год Квартал Месяц].CurrentMember
                ),
                [Measures].[Нарастающий итог ]
            )
        )'
SELECT
    Filter 
    (
        [NonFilterColumnSet],
        not IsEmpty([Measures].[Нарастающий итог]) 
    ) on columns,
    NON EMPTY 
    {
        Descendants 
        (
            [ОК].[ОКВЭД].[Все виды экономической деятельности],
            [ОК].[ОКВЭД].[Раздел],
             SELF 
        ) + 
        {
            [ОК].[ОКВЭД].[Все виды экономической деятельности]
        }   
    } on rows 
FROM [СТАТ_Инвестиции_Нефинансовые активы] 
WHERE 
    (
        [Инвестиции].[Нефинансовые активы].[Все показатели],
        [Группировки].[Инвестиции_Нефинансовые активы].[Все группировки].[По видам экономической деятельности],
        [Территории].[РФ].[Все территории].[Российская Федерация].[Уральский федеральный округ]<%subject_FO%>,
        [Источники данных].[Все источники данных].[СТАТ Отчетность - Субъект РФ]     
    ) 
  ]]>
