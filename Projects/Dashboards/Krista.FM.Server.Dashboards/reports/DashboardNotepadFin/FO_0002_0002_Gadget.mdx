-- FO_0002_0002_date 
with
    member [Measures].[ДанныеНа]
    as '[Период].[Период].CurrentMember.UniqueName'
    set [NonEmptySet]
    as 'Tail
        (
            Filter
            (
                [Период].[Период].Members,
                not IsEmpty([Measures].[Факт]) and
                [Период].[Период].CurrentMember.Name <> "Остатки на начало года" and
                [Период].[Период].CurrentMember.Name <> "Данные квартала" and
                [Период].[Период].CurrentMember.Name <> "Данные полугодия" and
                [Период].[Период].CurrentMember.Name <> "Данные года" and
                [Период].[Период].CurrentMember.Name <> "Данные месяца"
            ),
            1
        )'
select
    { [Measures].[ДанныеНа] } on columns,
    { [NonEmptySet] } on rows
from  [ФО_МесОтч_Доходы]
  ]]>
  </query>

    <![CDATA[
-- FO_0002_0002_incomes  
WITH           
    member Measures.[План]          
    as 'Measures.[Годовые назначения]'          
    member [Уровни бюджета].[СКИФ].[Бюджет гор.округа, мун.района и поселения]          
    as 'Aggregate          
        (
            {
                [Уровни бюджета].[СКИФ].[Бюджет поселения],
                [Уровни бюджета].[СКИФ].[Бюджет гор.округа],
                [Уровни бюджета].[СКИФ].[Бюджет района]          
            }          
        )'          
    member [Measures].[План ] 
    as ' 
        (
            [Measures].[Годовые назначения] / 1000000
        )' 
    member [Measures].[Факт ] 
    as ' 
        (
            [Measures].[Факт] / 1000000
        )' 
    member [Measures].[% исполнения ] 
    as ' 
        (
            [Measures].[% выполнения годовых назначений] 
        )' 
    member [Measures].[% темп роста ] 
    as ' 
        (
            [Measures].[Темп роста к аналогичному периоду предыдущего года_Факт] 
        )' 
    member [КД].[Сопоставимый].[Доходы ] 
    as ' 
        (
            [КД].[Сопоставимый].[Все коды доходов].[Доходы бюджета - Всего] 
        )'
    member  [Measures].[% исполнения прошлый год]  
    as ' 
        (
            [Период].[Период].[Данные всех периодов]<%period_last_date%>,
            [Measures].[% выполнения годовых назначений]  
        )'     
    member [Measures].[Факт прошлый год]
    as '
        (
            [Measures].[Факт_прошлый год] / 1000000
        )'
         member [КД].[Сопоставимый].[Первый ]
    as '1'
SELECT
    {
        [Measures].[План ] ,
        [Measures].[Факт ],
        [Measures].[% исполнения ],
        [Measures].[% исполнения прошлый год],
        [Measures].[Факт прошлый год],
        [Measures].[% темп роста ]
    } on columns,
    {
    [КД].[Сопоставимый].[Первый ],
        [КД].[Сопоставимый].[Доходы ]
    } on rows        
FROM  [ФО_МесОтч_Доходы]        
WHERE          
    (
        <%budget_level_enum%>,
        [ТипДокумента].[СКИФ].[Все].[Консолидированная отчетность и отчетность внебюджетных территориальных фондов],
        [Районы].[Сопоставимый].[Все районы].[Вологодская область],
        [Период].[Период].[Данные всех периодов]<%period_cur_date%>       
    )
    ]]>
  </query>
