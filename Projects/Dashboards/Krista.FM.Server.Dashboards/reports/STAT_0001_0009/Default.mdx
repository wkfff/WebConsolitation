<query id="STAT_0001_0009_grid">
<![CDATA[
-- STAT_0001_0009_grid
with
  member [Труд].[Трудовые ресурсы].[Среднегодовая численность занятого населения в <%period_year%> году, человек]
    as '[Труд].[Трудовые ресурсы].[Все показатели].[Среднегодовая численность занятого населения]'
    member [Труд].[Трудовые ресурсы].[Среднемесячная номинальная начисленная заработная плата и выплаты социального характера в расчете на одного работника в <%period_year%> году, руб.]
    as 'LookupCube
        (
            "[СТАТ_Труд_Заработная плата]",
            "
            (
                [Труд].[Заработная плата].[Все показатели].[Среднемесячная номинальная начисленная заработная плата и выплаты социального характера в расчете на одного работника],
                [Measures].[За период],
                [Территории].[РФ].[Все территории].[Российская Федерация].[Уральский федеральный округ]<%filter%>,
                " + MemberToStr
                (
                    [Период].[Год Квартал Месяц].CurrentMember
                ) + ",
                 " + MemberToStr
                (
                    [ОК].[ОКВЭД].CurrentMember
                ) + "
            )"
        )'
SELECT
    {
            [Труд].[Трудовые ресурсы].[Среднегодовая численность занятого населения в <%period_year%> году, человек],
            [Труд].[Трудовые ресурсы].[Среднемесячная номинальная начисленная заработная плата и выплаты социального характера в расчете на одного работника в <%period_year%> году, руб.]
        }
    ON COLUMNS ,
    NON EMPTY filter  
    (
        Descendants  
        (
            [ОК].[ОКВЭД].[Все виды экономической деятельности],
            [ОК].[ОКВЭД].[Раздел],
            SELF   
        ),
        not 
        (
            [ОК].[ОКВЭД].CurrentMember is [ОК].[ОКВЭД].[Все виды экономической деятельности].[Значение не указано]
        ) 
    ) ON ROWS
FROM [СТАТ_Труд_Трудовые ресурсы]
WHERE
    (
        [Группировки].[Труд_Трудовые ресурсы].[Все группировки].[По видам экономической деятельности] ,
        [Территории].[РФ].[Все территории].[Российская Федерация].[Уральский федеральный округ]<%subject_FO%>,
        [Measures].[За период],
        [Период].[Год Квартал Месяц].[Данные всех периодов].[<%period_year%>]
    )
]]>
</query>
